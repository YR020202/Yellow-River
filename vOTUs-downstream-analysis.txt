#Step 4: Filter taxonomic classification results to retain highest-scoring hits per contig
#tax-filter.py script processes CAT classification output to select best matches
#This script reads taxonomic classification results and retains only the top-scoring hit for each contig
#Import required Python libraries
import csv  # For reading and writing tab-delimited files
import sys  # For system operations and error handling
from collections import defaultdict # For creating dictionaries with default values


def process_and_write(input_file, output_file):
# Dictionary to store contig groups, keyed by contig prefix
    groups = defaultdict(list)
    header = None

# Read input file
    with open(input_file, 'r', newline='', encoding='utf-8') as f:
        lines = f.readlines()
        if not lines:
            print("none")
            return

# Parse header line (assumes tab-separated format)
        header = lines[0].strip().split('\t')
        if header[0] != '# ORF':
            raise ValueError("First column of input file must be '# ORF'")

# Process each classification record
        for line in lines[1:]:
            line = line.strip()
            if not line:
                continue

# Parse tab-separated values into dictionary
            parts = line.split('\t')
            row = {header[i]: parts[i] for i in range(len(parts))}
            print(row)

# Extract ORF identifier
            orf = row['# ORF']

# Extract contig prefix from ORF identifier
# Format: contigID_ORFnumber
            if 'ORF has no hit to database' in orf:
                prefix = orf
            else:
                parts_orf = orf.split('_', 2)
                prefix = parts_orf[0] + '_' + parts_orf[1]

# Extract bit-score for ranking (higher score = better match)
            try:
                if 'top bit-score' in row:
                    score = row['top bit-score']
                    numeric_score = float(score) if '.' in score else int(score)
                else:
                    numeric_score = None
            except ValueError:
                numeric_score = None
# Store row with its contig prefix
            groups[prefix].append((row, numeric_score))
# Select best hit for each contig
    result = []
    for prefix, rows in groups.items():
# Filter rows with valid scores
        valid_rows = [(r, s) for r, s in rows if s is not None]
        if valid_rows:
# Sort by score in descending order and select top hit
            valid_rows.sort(key=lambda x: x[1], reverse=True)
            result.append(valid_rows[0][0])
        else:
# If no valid scores, select first row
            result.append(rows[0][0])
# Write filtered results to output file
    with open(output_file, 'w', newline='', encoding='utf-8') as f:
        writer = csv.DictWriter(f, fieldnames=header, delimiter='\t')
        writer.writeheader()
        for row in result:
            writer.writerow(row)
