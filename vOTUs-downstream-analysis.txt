#Identify viral protein clusters (vPCs) from vOTUs
#Step 1: Predict protein-coding genes from vOTUs
#prodigal predicts open reading frames and translates protein sequences
#-i specifies the input FASTA file containing viral genome sequences (vOTUs)
#-a specifies the output protein sequence FASTA file (vOTUs.faa)
#-d specifies the output nucleotide sequence FASTA file for predicted genes (vOTUs.fna)
#-p meta indicates the prediction mode optimized for metagenomic sequences
prodigal -i path/to/vOTUs.fasta -a vOTUs.faa -d  vOTUs.fna  -p meta

#Step 2: Cluster predicted viral proteins into protein clusters using CD-HIT
#cd-hit performs sequence clustering to group similar proteins and reduce redundancy
#-c 0.6 sets the sequence identity threshold for clustering (60% identity)
#-aS 0.8 sets the alignment coverage threshold for the shorter sequence (80% coverage)
#-n 4 specifies the word length for fast clustering (appropriate for 60% identity threshold)
#-g 1 enables accurate clustering mode (slow but more accurate)
#-i specifies the input protein sequence FASTA file
#-o specifies the output file containing representative sequences of protein clusters
cd-hit -c 0.6 -aS 0.8 -n 4 -g 1 -i path/to/vOTU.faa -o path/to/vPCs.faa

#Identify auxiliary metabolic genes (AMGs) in viral genomes
#emapper.py performs functional annotation using eggNOG database
#-m diamond specifies the alignment method (DIAMOND for fast protein alignment)
#-i specifies the input FASTA file containing predicted viral proteins
#--cpu 40 specifies the number of CPU threads to use for parallel processing
#--data_dir specifies the directory containing eggNOG database files
#--dmnd_db specifies the DIAMOND-formatted eggNOG protein database for alignment
#--output_dir specifies the directory where annotation results will be stored
#-o vOTU specifies the base name for output files
#--evalue 0.00001 sets the E-value cutoff for significant matches (1e-5)
#--score 50 sets the minimum bit score threshold for significant matches
#This command annotates viral proteins to identify metabolic genes that may influence host metabolism
emapper.py \
-m diamond -i path/to/vOTU_proteins.faa --cpu 40 \
--data_dir path/to/eggnogdb2 \
--dmnd_db path/to/eggnogdb2/eggnog_proteins.dmnd \
--output_dir /home/metagenomes/LILU/S_ALL_Test/blastdb/eggnog -o vOTU --evalue 0.00001 --score 50

# -o specifies the output directory for all METABOLIC result files, including functional profiles and pathway summaries

# Quantify vOTU abundance profiles using CoverM
# coverm contig calculates coverage metrics for each vOTU based on read mapping
# --methods trimmed_mean specifies the trimmed mean coverage (default 10% trim) to reduce bias from outlier positions
# --coupled provides the paired-end clean read files (R1 and R2) for read mapping
# --reference points to the FASTA file containing all vOTU reference sequences
# -o defines the output file path for the resulting abundance matrix (TSV format)
# --threads 40 enables parallel processing using 40 CPU cores
# This command generates reliable vOTU relative abundance estimates for downstream ecological analyses
coverm contig --methods trimmed_mean --coupled Clean_reads_R1.fq Clean_reads_R2.fq --reference vOTU.fasta -o Abundance_sample_ID.tsv --threads 40

# Calculate vOTU coverage breadth using CoverM
# coverm contig computes the fraction of each vOTU reference covered by mapped reads
# --methods covered_fraction specifies the coverage breadth metric (proportion of bases with ≥1 read)
# --coupled provides the paired-end clean read files (R1 and R2) for read mapping
# --reference points to the FASTA file containing all vOTU reference sequences
# -o defines the output file path for the resulting breadth matrix (TSV format)
# --threads 40 enables parallel processing using 40 CPU cores
# This command determines viral presence (coverage breadth ≥70%) and filters false positives from cross‑mapping noise
coverm contig --methods covered_fraction --coupled Clean_reads_R1.fq Clean_reads_R2.fq --reference vOTU.fasta  -o Fraction_sample_ID.tsv --threads 40
